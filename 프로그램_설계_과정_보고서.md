    """

    dC_dx = torch.autograd.grad(C, x[:, 1:])  # κ³µκ°„ λ―Έλ¶„ (2D/3D)
    d2C_dx2 = torch.autograd.grad(dC_dx, x)   # 2μ°¨ λ―Έλ¶„ (ν™•μ‚°ν•­)

    residual = dC_dt - D * d2C_dx2 + v * dC_dx + k_decay * C

    pde_loss = (residual ** 2).mean()

    return pde_loss
